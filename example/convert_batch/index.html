#include <thread>
#include <vector>

void ConvertBatchParallel(const std::vector<std::string> &files, int numThreads)
{
    std::vector<std::thread> threads;
    size_t filesPerThread = files.size() / numThreads;

    for (int i = 0; i < numThreads; ++i) {
        size_t start = i * filesPerThread;
        size_t end = (i == numThreads - 1) ? files.size() : start + filesPerThread;

        threads.emplace_back([=]() {
            AmrtConverter converter;
            std::vector<std::string> batch(files.begin() + start, files.begin() + end);
            converter.ConvertBatch(batch);
        });
    }

    for (auto &t : threads) {
        t.join();
    }
}

int ConvertWithRetry(AmrtConverter &converter, 
                     const std::vector<std::string> &files, 
                     int maxRetries = 3)
{
    std::vector<std::string> failedFiles;

    for (const auto &file : files) {
        bool success = false;

        for (int retry = 0; retry < maxRetries && !success; ++retry) {
            Result<void> result = converter.Convert(file);
            success = result.IsSuccess();

            if (!success && retry < maxRetries - 1) {
                std::cout << "é‡è¯• " << file << " (ç¬¬" << (retry + 1) << "æ¬¡)" << std::endl;
            }
        }

        if (!success) {
            failedFiles.push_back(file);
        }
    }

    return files.size() - failedFiles.size();
}
